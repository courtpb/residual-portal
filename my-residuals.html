// Render chart with Y-axis steps of 100,000 function renderChart(){ const ctx = document.getElementById('residualChart').getContext('2d'); const visibleResiduals = sortResiduals(getVisibleResiduals()); const labels = visibleResiduals.map(r=>r.month).reverse(); // oldest to newest const data = visibleResiduals.map(r=>r.amount).reverse(); new Chart(ctx, { type: 'bar', data: { labels: labels, datasets: [{ label: 'Residual Income', data: data, backgroundColor: 'rgba(0,115,170,0.7)', borderColor: 'rgba(0,115,170,1)', borderWidth: 1 }] }, options: { responsive:true, scales: { y: { beginAtZero: true, ticks: { stepSize: 100000, callback: function(value){return "$" + value.toLocaleString();} } } } } }); }
