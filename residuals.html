function importFile(){ let file = document.getElementById("fileUpload").files[0] let monthInput = document.getElementById("residualMonth").value let bucket = document.getElementById("bucket").value if(!file){ alert("Please upload a file"); return } if(!monthInput){ alert("Please select a month/year"); return } let parts = monthInput.split("-") let selectedMonthYear = parts[1] + "/" + parts[0] let reader = new FileReader() reader.onload = function(e){ let data = new Uint8Array(e.target.result) let workbook = XLSX.read(data,{type:"array"}) let sheet = workbook.Sheets[workbook.SheetNames[0]] let range = XLSX.utils.decode_range(sheet['!ref']) let merchants = [] // --- HEADER MAP --- let headerMap = {} for(let C=range.s.c; C<=range.e.c; C++){ let addr = XLSX.utils.encode_cell({r:0,c:C}) let cell = sheet[addr] if(cell && cell.v){ headerMap[cell.v.toString().toLowerCase().trim()] = C } } let residualCol = headerMap['residual'] let merchantCol = headerMap['merchant number'] let dbaCol = headerMap['dba'] let repCol = headerMap['sales rep id'] if(residualCol === undefined){ alert("Could not find 'Residual' column.") return } // 🔥 SALES + RETURN COLUMN NAMES const salesCols = [ 'mastercard','visa','discover_acquiring','amex_opt_blue', 'ebt','accel debit','interlink debit','maestro debit','pulse debit','star debit' ]; const returnCols = [ 'mastercard return','visa return','discover_acquiring return','amex_opt_blue return' ]; for(let R=2; R<=range.e.r; ++R){ let getCell = (colIdx)=>{ let addr = XLSX.utils.encode_cell({r:R,c:colIdx}) return sheet[addr] ? sheet[addr].v : 0 } let merchant = getCell(merchantCol) if(!merchant) continue let dba = getCell(dbaCol) let repId = getCell(repCol) let residualVal = getCell(residualCol) let residual = !isNaN(residualVal) ? Number(residualVal) : 0 // 🔥 CALCULATE SALES let salesVolume = 0 salesCols.forEach(col=>{ let idx = headerMap[col] if(idx !== undefined){ let val = getCell(idx) if(!isNaN(val)) salesVolume += Number(val) } }) // 🔥 CALCULATE RETURNS let returnVolume = 0 returnCols.forEach(col=>{ let idx = headerMap[col] if(idx !== undefined){ let val = getCell(idx) if(!isNaN(val)) returnVolume += Number(val) } }) let processingVolume = salesVolume - returnVolume merchants.push({ monthYear: selectedMonthYear, merchant: merchant, dba: dba, repId: repId, total: residual, volume: processingVolume }) } let batches = JSON.parse(localStorage.getItem("residualBatches")) || [] batches.push({ batchId: Date.now(), month: selectedMonthYear, bucket: bucket, merchants: merchants }) localStorage.setItem("residualBatches", JSON.stringify(batches)) loadBatches() alert("Residual file imported successfully") } reader.readAsArrayBuffer(file) }
